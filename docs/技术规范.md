- **组件引用**: PascalCase (如 `<MediaCard />`)

```typescript
// composables/useEmby.ts
import { ref, readonly } from 'vue'

export function useEmby() {
  const media = ref<MediaItem | null>(null)
  const loading = ref(false)
  const error = ref<string | null>(null)

  const fetchMedia = async (id: string) => {
    loading.value = true
    error.value = null
    
    try {
      media.value = await embyApi.getMedia(id)
    } catch (err) {
      error.value = err.message
    } finally {
      loading.value = false
    }
  }

  return {
    media: readonly(media),
    loading: readonly(loading),
    error: readonly(error),
    fetchMedia
  }
}
```

## API规范

### Emby API封装

#### API结构
```typescript
// services/emby-api.ts
class EmbyApi {
  private baseUrl: string
  private apiKey: string
  private userId: string

  constructor(config: EmbyConfig) {
    this.baseUrl = config.serverUrl
    this.apiKey = config.apiKey
    this.userId = config.userId
  }

  // 获取媒体库
  async getLibraries(): Promise<MediaLibrary[]> {
    return this.request('/Library/VirtualFolders')
  }

  // 获取媒体项
  async getMediaItems(params: MediaQueryParams): Promise<MediaItem[]> {
    return this.request('/Users/{UserId}/Items', {
      params: {
        ...params,
        UserId: this.userId
      }
    })
  }

  // 通用请求方法
  private async request<T>(endpoint: string, options?: RequestOptions): Promise<T> {
    const url = `${this.baseUrl}${endpoint}`
    const response = await fetch(url, {
      ...options,
      headers: {
        'X-Emby-Token': this.apiKey,
        ...options?.headers
      }
    })
    
    if (!response.ok) {
      throw new EmbyApiError(response.status, response.statusText)
    }
    
    return response.json()
  }
}
```

```typescript
// stores/emby.ts
import { defineStore } from 'pinia'

interface EmbyState {
  server: EmbyServer | null
  libraries: MediaLibrary[]
  currentMedia: MediaItem | null
  loading: boolean
  error: string | null
}

export const useEmbyStore = defineStore('emby', {
  state: (): EmbyState => ({
    server: null,
    libraries: [],
    currentMedia: null,
    loading: false,
    error: null
  }),

  getters: {
    isConnected: (state) => !!state.server,
    hasLibraries: (state) => state.libraries.length > 0
  },

  actions: {
    async connect(config: EmbyConfig) {
      this.loading = true
      this.error = null
      
      try {
        this.server = await embyApi.connect(config)
        await this.fetchLibraries()
      } catch (err) {
        this.error = err.message
        throw err
      } finally {
        this.loading = false
      }
    },

    async fetchLibraries() {
      this.libraries = await embyApi.getLibraries()
    }
  }
})
```

```typescript
// services/players/mpv-player.ts
class MpvPlayer implements Player {
  name = 'MPV'
  path: string
  isInstalled = false

  async play(url: string, options?: PlayOptions): Promise<void> {
    const args = this.buildArgs(url, options)
    await this.spawn(args)
  }

  private buildArgs(url: string, options?: PlayOptions): string[] {
    const args = [url]
    
    if (options?.fullscreen) {
      args.push('--fullscreen')
    }
    
    if (options?.hardwareDecoding) {
      args.push('--hwdec=auto')
    }
    
    return args
  }
}
```

### 缓存服务

```typescript
// services/cache-service.ts
class CacheService {
  private config: CacheConfig
  private index: Map<string, CacheItem> = new Map()

  async get(key: string): Promise<string | null> {
    const item = this.index.get(key)
    if (!item) return null

    // 更新访问时间
    item.lastAccessed = new Date()
    await this.saveIndex()

    return item.path
  }

  async set(key: string, data: Buffer): Promise<string> {
    // 检查缓存大小
    await this.cleanup()

    // 保存文件
    const path = this.generatePath(key)
    await fs.writeFile(path, data)

    // 更新索引
    const item: CacheItem = {
      key,
      path,
      size: data.length,
      lastAccessed: new Date(),
      created: new Date()
    }
    this.index.set(key, item)
    await this.saveIndex()

    return path
  }

  private async cleanup(): Promise<void> {
    const totalSize = this.getTotalSize()
    if (totalSize <= this.config.maxSize) return

    // 按最后访问时间排序，删除最旧的
    const items = Array.from(this.index.values())
      .sort((a, b) => a.lastAccessed.getTime() - b.lastAccessed.getTime())

    while (this.getTotalSize() > this.config.maxSize && items.length > 0) {
      const item = items.shift()!
      await this.remove(item.key)
    }
  }
}
```

```typescript
export function handleError(error: unknown): AppError {
  if (error instanceof AppError) {
    return error
  }

  if (error instanceof Error) {
    return new AppError('UNKNOWN_ERROR', error.message, { original: error })
```

## 版本控制规范

### 分支策略
- `main`: 主分支，稳定版本
- `develop`: 开发分支
- `feature/*`: 功能分支
- `hotfix/*`: 紧急修复分支

### Git提交
```bash
# 功能开发
git checkout -b feature/emby-connection
git commit -m "feat(emby): 实现Emby服务器连接功能"

# Bug修复
git checkout -b fix/player-crash
git commit -m "fix(player): 修复播放器崩溃问题"
```
